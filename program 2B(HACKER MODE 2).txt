#include <bits/stdc++.h>

using namespace std;
# define ll long long int
int main()
{
    ll n,q,i,ans=-1;
    cin>>n>>q;
    std::vector<pair<ll,ll>> vec ;
    
    while(q--){
        pair<ll,ll> p1;
        pair<ll,ll> p2;
        cin>>p1.first>>p2.first>>p1.second;
        p2.second=p1.second*-1;
        vec.push_back(p1);
        vec.push_back(p2);
        
    }
    
    sort(vec.begin(),vec.end());
    
    for(i=1;i<vec.size();i++)       //prefix sum
     vec[i].second+=vec[i-1].second;
    
    for(i=0;i<vec.size();i++)         //adding i to a[i]
    { vec[i].second+=(i+1);
           ans=max(ans,vec[i].second);
    }
    ans=max(n,ans);
    cout<<ans;
   
    return 0;
}
